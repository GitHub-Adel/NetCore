(SOLUTION EXPLORER)(SocialMedia.Api/Controllers)
--------------------------------
se comunica with views a traves de la interfaces

1) define la Route
2) Inherited ControllerBase
3) inject la interface to use su methods.
4) create los methods with su Http verb.
5) alway return any code. Ok() = 200


[Route("api/[Controller]")]
[ApiController]
public class NameController:ControllerBase
{
	private readonly INameRepository nameRepository;
	public NameRepository(INameRepository nameRepository)
	{
		this.nameRepository = nameRepository;
	}

	[HttpGet]
	public IActionResult GetPost()
	{
		var post = nameRepository.GetPost();
		return Ok( post )
	}
}
