namespace SocialMedia.Infrastructure.Validator
{
    public class UserValidator : AbstractValidator<UserDTO>
    {
        public UserValidator()
        {
            RuleFor(x => x.Firstname).NotEmpty().WithMessage("El Firstname no puede quedar vacío");
        }
    }
}
